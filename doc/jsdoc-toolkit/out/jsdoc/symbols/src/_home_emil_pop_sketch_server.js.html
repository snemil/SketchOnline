  6 /**
  7 * Server - Back-end
  8 * @default 
  9 * @param clients array with all users connected to server.
 10 * @param oldX - coordinate
 11 * @param oldY - coordinate
 12 */
 13 
 14 var io = require('socket.io').listen(8080);
 15 var clients = [];
 16 var oldX = 0;
 17 var oldY = 0;
 18 
 19 
 20 io.sockets.on('connection', function(socket){
 21 	
 22 	console.log('New client connected ' + socket);
 23 	socket.on('client_connected', function(data){
 24 		data.id = socket.id;
 25 		console.log('INNE I CLIENT CONNECTED');
 26 		clients.push(data);
 27 		io.sockets.emit("send_data", clients);	
 28 	});
 29 
 30 	socket.on('drawClick', function(player){
 31 		for(client in clients){
 32 			if(client.id === socket.id){
 33 				client.x = player.x;
 34 				client.y =  player.y;
 35 				client.type = player.type;
 36 				console.log("Client: " + socket.id);
 37 				console.log("X: " + player.x + ", Y:" + player.y);
 38 				break;
 39 			}
 40 
 41 			}
 42 			
 43 		socket.broadcast.emit("draw", {
 44 			id: player.id,
 45 			x: player.x,
 46 			y: player.y,
 47 			oldX: oldX,
 48 			oldY: oldY,
 49 			type: player.type
 50 		});
 51 		oldX = player.x;
 52 		oldY = player.y;
 53 	});
 54 
 55 	socket.on('newLine', function(player){
 56 			oldX = player.x;
 57 			oldY = player.y;
 58 	});
 59 
 60 	socket.on('disconnect', function(){
 61 		for(i = 0; i < clients.length; i++){
 62 			if(clients[i].id === socket.id){
 63 				clients.splice(i,1);
 64 				break;
 65 			}
 66 		}	
 67 		console.log(clients.length);
 68 		console.log("Client: " + socket.id + " disconnected");
 69 		socket.broadcast.emit("send_data", clients);	
 70 	});
 71 });
