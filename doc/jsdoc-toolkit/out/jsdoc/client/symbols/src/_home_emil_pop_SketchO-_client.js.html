  2 * @fileOverview

  5 * This is the client-side application. A users loads this page and gets connected to the server.
  6 */
  7  
  8 /**
  9 * Client side
 10 * @namespace CLIENT
 11 * @class CLIENT
 12 */ 
 13 
 14 
 15 $(document).ready(function(){
 16 
 17 /**
 18 * clients variable for storing clients socketID 
 19 * @type array
 20 */
 21 	var clients = [];
 22 
 23 /**
 24 * canvas the canvas element 
 25 * @type HTML5/canvas
 26 */
 27 	var canvas = document.getElementById("can");
 28 
 29 /**
 30 * ctx HTML5/canvas variable 
 31 * @type HTML5/canvas
 32 */
 33 	var ctx = canvas.getContext("2d");
 34 	ctx.strokeStyle = 'black';
 35 	ctx.lineCap ="round";
 36 	ctx.lineJoin= "round";
 37 	ctx.lineWidth = 10;
 38 
 39 /**
 40 * worker JS-thread variable. {@link worker} 
 41 * @type JS-thread
 42 */
 43 	var worker = new Worker('scripts/worker.js');
 44 
 45 
 46 /**
 47 * player variable containing player characteristics 
 48 * @type array
 49 */
 50 	var player = {
 51 		color: 'black',
 52 		size: '15',
 53 		x: 0,
 54 		y: 0,
 55 		oldX: 0,
 56 		oldY: 0,
 57 		flag: true
 58 	};
 59 
 60 /**
 61 * socket websocket for connection to server 
 62 * @type websocket
 63 */
 64 	var socket = io.connect('http://192.168.1.136:8080');
 65 
 66 
 67 /**
 68 * @param 'event'
 69 * @param callback-function
 70 * 'connect' event: client is connected to server.
 71 * 'new_client' event: adds the new client to html-page
 72 * 'client_dc' event: removes the new client to html-page.
 73 * 'previous_drawn' event: draws the previously drawn canvas.
 74 * 'draw' event: calls draw-function
 75 * 'resetWorkspace' event: calls the resetWorkspace function
 76 * 'draw drawstart drawend' event: calls drawfunction and sends data.
 77 * @name on
 78 * @function
 79 * @returns void
 80 */
 81     socket.on('connect', function() {
 82         socket.emit('client_connected', {id: 0, color: player.color, size: player.size});
 83       });
 84 
 85 	socket.on('new_client', function(send_clients){
 86 		if ($('.client_item').length != 0) {
 87 			$('#clients').append('<li class="client_item">'+ 'Client id: ' + send_clients[send_clients.length -1].id +
 88 			' is connected</li>');
 89 		}else {
 90 			for (var i = 0; i<send_clients.length; i++) {
 91 				$('#clients').append('<li class="client_item">' + 'Client id: ' + send_clients[i].id +
 92 				' is connected</li>');
 93 			}
 94 		}
 95 		clients = send_clients;
 96 	});
 97 
 98 	socket.on('client_dc', function(client_dc){
 99 		$('.client_item').each(function(){
100 			if ('Client id: ' + client_dc.id +' is connected' == $(this).text()){
101 				$(this).remove();
102 			}
103 		});
104 	});
105 
106 	socket.on('previous_drawn', function(lines) {
107 		for (var i = 0; i<lines.length; i++) {
108 			draw(lines[i].x, lines[i].y, lines[i].oldX, lines[i].oldY, lines[i].color, lines[i].size);
109 		}
110 	});
111 
112 	socket.on('draw', function(data){
113 		worker.postMessage({
114 			id: data.id,
115 			x: data.x,
116 			y: data.y,
117 			oldX: data.oldX,
118 			oldY: data.oldY
119 		});
120 
121 		draw(data.x, data.y, data.oldX, data.oldY, data.color, data.size);
122 	});
123 
124 	socket.on('resetWorkspace', function() {
125 		resetWorkspace();
126 	});
127 
128 /**
129 * This method .. message event for worker.
130 * @param 'event' 
131 * 'message' event: waits for message from worker
132 * @name addEventListener
133 * @function
134 */
135 	worker.addEventListener('message', function(e) {
136 		var found = false;
137 		$('#total').text(e.data.sum);
138 		$('.pixel_dist').each(function(){
139 			var pixID = $(this).find('.id');
140 			if (pixID.text() == e.data.id){
141 				$(this).find('.px').text(e.data.distance);
142 				found = true;
143 			}
144 		});
145 		if(!found){
146 			$('#pixels').append('<li class="pixel_dist">' + 'Client: <span class = "id">' + e.data.id + '</span>'+
147 			 ' has drawn ' + '<span class="px">' + e.data.distance + '</span> pixels' + '</li>');
148 		}
149 	});
150 
151 /**
152 * This function clears the canvas
153 * @name resetWorkspace
154 * @function
155 * @returns void
156 */
157 	function resetWorkspace() {
158 		ctx.rect(0, 0, 1000, 800);
159 		ctx.fillStyle = 'white';
160 		ctx.fill();
161 	}
162 /**
163 * This method draws on canvas
164 * @param x current x-value
165 * @param y current x-value
166 * @param oldX old x-value
167 * @param oldY old y-value
168 * @param color current color
169 * @size current size
170 * @name draw 
171 * @function
172 * @returns void
173 */
174 	function draw(x, y, oldX, oldY, color, size) {
175 	      	ctx.strokeStyle = color;
176 	      	ctx.lineWidth = size;
177 	        ctx.beginPath();
178 	        ctx.moveTo(oldX, oldY);
179 	        ctx.lineTo(x, y);
180 	        ctx.stroke();
181 	        ctx.closePath();
182 	} 
183 	
184 	$('#can').on('drag dragstart dragend', function(e){
185 		var x, y;
186 		x = (e.clientX - this.offsetLeft);
187 		y = (e.clientY - this.offsetTop);
188 	
189 		draw(x, y, player.oldX, player.oldY, player.color, player.size);
190 
191 /**
192 * This function sends data to other clients
193 * @name emit
194 * @param 'event'
195 * 'drawClick' event: sends drawn data.
196 * @function
197 * @returns void
198 */
199 		socket.emit('drawClick', {
200 			x: x,
201 			y: y,
202 			oldX: player.oldX,
203 			oldY: player.oldY,
204 			color: player.color,
205 			size: player.size
206 		});
207 		player.oldX = x;
208 		player.oldY = y;
209 	});
210 
211 	$('#can').on('mousedown', function(e){
212 		player.oldX = (e.clientX - this.offsetLeft);
213 		player.oldY = (e.clientY - this.offsetTop);
214 
215 
216 	});
217 
218 	$('#button_black').on('click', function(e){
219 		player.color = 'black';
220 	});
221 
222 	$('#button_blue').on('click', function(e){
223 		player.color = 'blue';
224 	});
225 
226 	$('#button_red').on('click', function(e){
227 		player.color = 'red';
228 	});
229 
230 	$('#button_yellow').on('click', function(e){
231 		player.color = 'yellow';
232 	});
233 
234 	$('#button_green').on('click', function(e){
235 		player.color = '00FF00';
236 	});
237 
238 	$('#button_white').on('click', function(e){
239 		player.color = 'white';
240 	});
241 
242 	$('#button_size1').on('click', function(e){
243 		player.size = '3';
244 	});
245 
246 	$('#button_size2').on('click', function(e){
247 		player.size = '8';
248 	});
249 
250 	$('#button_size3').on('click', function(e){
251 		player.size = '15';
252 	});
253 
254 	$('#button_size4').on('click', function(e){
255 		player.size = '25';
256 	});
257 
258 	$('#button_reset').on('click', function(e){
259 		socket.emit('resetWorkspace', {});
260 		resetWorkspace();
261 	});
262 });
